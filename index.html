מה יקרה אם ננסה להריץ את הקוד הבא?

const arr1 = [1, 2, 3];
const arr2 = [1, 2, 3];
console.log(arr1 === arr2);

const obj1 = {name: "Alex"};
const obj2 = {name: "Alex"};

א) true, true - אותם ערכים

ב) false, false - מערכים ואובייקטים שונים

ג) true, false - מערכים שווים, אובייקטים לא

ד) תלוי במצב

✅ תשובה נכונה: ב (false, false)

הסבר המפתיע:

JavaScript משווה מערכים ואובייקטים לפי הפניה (reference) ולא לפי ערך!

כל מערך/אובייקט נמצא במיקום שונה בזיכרון, אז הם לא שווים גם אם התוכן זהה.

// הבעיה:
const arr1 = [1, 2, 3];
const arr2 = [1, 2, 3];
console.log(arr1 === arr2);  // false 😱

// אותו דבר באובייקטים:
const obj1 = {name: "Alex"};
const obj2 = {name: "Alex"};
console.log(obj1 === obj2);  // false 😱

// אבל אם מפנים לאותו מקום:
const arr3 = arr1;
console.log(arr1 === arr3);  // true ✅

💡 איך כן משווים?

// דרך 1: JSON.stringify (פשוטה)

// דרך 2: לולאה (לפשוטים)
function arraysEqual(a, b) {
  if (a.length !== b.length) return false;
  for (let i = 0; i < a.length; i++) {
    if (a[i] !== b[i]) return false;
  }
  return true;
}
console.log(arraysEqual(arr1, arr2));  // true ✅

// דרך 3: Object.keys (לאובייקטים פשוטים)
function objectsEqual(obj1, obj2) {
  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);
  
  if (keys1.length !== keys2.length) return false;
  
  for (let key of keys1) {
    if (obj1[key] !== obj2[key]) return false;
  }
  return true;
}

❓ Undefined (שכחנו להגדיר)

מה זה? כשיצרתם משתנה אבל שכחתם לשים בו משהו - זה undefined

🎯 דוגמאות בסיסיות:

let name;           // undefined
let age;            // undefined
let address;        // undefined

console.log(name);  // undefined

🔥 דברים מעניינים:

1. מתי מקבלים undefined? 
// משתנה לא מאותחל
let x;
console.log(x);  // undefined

// פרמטר חסר
function greet(name) {
  console.log(name);
}
greet();  // undefined

// פרופרטי שלא קיים
let obj = {a: 1};
console.log(obj.b);  // undefined

// פונקציה בלי return
function test() {}
console.log(test());  // undefined
2. בדיקת undefined: 
let x;

x === undefined;        // true ✅
typeof x === "undefined"; // true ✅

// ⚠️ היזהר:
x == null;  // true (!!!)
3. טיפים: 
// ערכי ברירת מחדל
function greet(name = "אורח") {
  console.log("שלום " + name);
}
greet();  // "שלום אורח"

// Optional Chaining
let user = {};
console.log(user?.address?.city);  // undefined
// במקום: user.address.city (שגיאה!)

// Nullish Coalescing
let x;
let result = x ?? "ברירת מחדל";
console.log(result);  // "ברירת מחדל"

למה צריך רקורסיה?

יתרונות:

  • קוד קצר ואלגנטי לבעיות מורכבות
  • מושלם למבני נתונים מקוננים (עצים, תיקיות)
  • פותר בעיות שקשה לפתור עם לולאה רגילה
  • דוגמאות: חיפוש בעץ, Fibonacci, מיון Merge Sort

בעיות אפשריות:

  • ביצועים: רקורסיה איטית יותר מלולאה (overhead של קריאות)
  • זיכרון: כל קריאה נשמרת ב-call stack - יכול להיגמר המקום
  • Stack Overflow: יותר מדי קריאות רקורסיביות גורמות לקריסה
  • קשה לדיבוג: צריך לעקוב אחרי הרבה קריאות

כלל אצבע: השתמשו ברקורסיה כשהבעיה באמת רקורסיבית. אחרת - לולאה פשוטה עדיפה!
